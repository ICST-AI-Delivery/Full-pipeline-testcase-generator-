def analyze_fpi(image_path, fpi_data):
    """
    Master script to route FPI analysis based on diagram type
    """
    
    # Step 1: Initial Assessment
    diagram_type = identify_diagram_type(image_path)
    complexity_score = assess_complexity(image_path)
    
    # Step 2: Route to Appropriate Analysis Method
    if diagram_type == "SEQUENTIAL_FLOW":
        return analyze_sequential_flow(image_path, fpi_data)
    elif diagram_type == "STATE_MATRIX":
        return analyze_state_matrix(image_path, fpi_data)
    elif diagram_type == "HYBRID":
        return analyze_hybrid_diagram(image_path, fpi_data)
    else:
        return analyze_unknown_type(image_path, fpi_data)

def identify_diagram_type(image_path):
    """
    Determine diagram type based on visual elements
    """
    # Check for terminators (start/end ovals)
    has_terminators = check_for_terminators(image_path)
    
    # Check for sequential blocks
    has_sequential_blocks = check_for_sequential_blocks(image_path)
    
    # Check for matrix pattern
    has_matrix_pattern = check_for_matrix_pattern(image_path)
    
    # Decision logic
    if has_terminators and has_sequential_blocks:
        return "SEQUENTIAL_FLOW"
    elif has_matrix_pattern and not has_terminators:
        return "STATE_MATRIX"
    elif has_terminators and has_matrix_pattern:
        return "HYBRID"
    else:
        return "UNKNOWN"

def assess_complexity(image_path):
    """
    Calculate complexity score for processing time estimation
    """
    block_count = count_blocks(image_path)
    decision_count = count_decision_points(image_path)
    cycle_count = detect_cycles(image_path)
    
    complexity_score = (decision_count * 2) + (block_count * 1) + (cycle_count * 3)
    
    return {
        'score': complexity_score,
        'level': get_complexity_level(complexity_score),
        'estimated_time': calculate_processing_time(block_count, decision_count, cycle_count),
        'blocks': block_count,
        'decisions': decision_count,
        'cycles': cycle_count
    }
```

def analyze_sequential_flow(image_path, fpi_data):

    Complete sequential flow diagram analysis
    """
    
    # Step 1: Extract all blocks
    blocks = extract_blocks_from_image(image_path)
    
    # Step 2: Analyze each block
    block_analysis = []
    for i, block in enumerate(blocks):
        analysis = analyze_single_block(block, i+1)
        block_analysis.append(analysis)

    flow_paths = map_flow_paths(blocks, block_analysis)
    
    # Step 4: Detect cycles
    cycles = detect_cyclic_patterns(flow_paths)
    
    # Step 5: Extract decision points
    decision_points = extract_decision_logic(blocks)
    
    # Step 6: Build state matrix
    state_matrix = build_sequential_state_matrix(block_analysis, decision_points)

        'diagram_type': 'SEQUENTIAL_FLOW',
        'blocks': block_analysis,
        'flow_paths': flow_paths,
        'cycles': cycles,
        'decision_points': decision_points,

def analyze_state_matrix(image_path, fpi_data):
    """
    Traditional state matrix analysis
    """
    
    # Step 1: Identify matrix structure
    matrix_structure = identify_matrix_structure(image_path)

    input_parameters = extract_input_parameters(image_path)
    
    # Step 3: Extract output parameters
    output_parameters = extract_output_parameters(image_path)
    
    # Step 4: Identify initial state
    initial_state = identify_initial_state(image_path)
    
    # Step 5: Build complete state matrix
    state_matrix = build_complete_state_matrix(
        input_parameters, 
        output_parameters, 
        initial_state
    )
    
    # Step 6: Validate matrix completeness
    validation = validate_state_matrix(state_matrix, input_parameters, output_parameters)
    
    return {
        'diagram_type': 'STATE_MATRIX',
        'matrix_structure': matrix_structure,
        'input_parameters': input_parameters,
        'output_parameters': output_parameters,
        'initial_state': initial_state,
        'state_matrix': state_matrix,
        'validation': validation,
        'complexity': assess_matrix_complexity(state_matrix)
    }

def build_complete_state_matrix(input_params, output_params, initial_state):
    """
    Generate all possible input/output combinations
    """
    import itertools
    
    # Generate all combinations of input values
    input_combinations = list(itertools.product(
        *[param['values'] for param in input_params]
    ))
    
    state_matrix = []
    
    for combination in input_combinations:
        # Build input state
        input_state = {}
        for i, param in enumerate(input_params):
            input_state[param['name']] = combination[i]
        
        # Add initial state
        input_state['initial_state'] = initial_state
        
        # Calculate output state
        output_state = calculate_output_state(input_state, output_params)
        
        # Create matrix row
        matrix_row = {
            'inputs': input_state,
            'outputs': output_state,
            'description': generate_state_description(input_state, output_state)
        }
        
        state_matrix.append(matrix_row)
    
    return state_matrix
```

def process_can_signals(text_content):
    """
    Extract and interpret automotive CAN signals
    """
    import re
    
    # Pattern for CAN signal format: DOMAIN.SIGNAL_NAME.Parameter
    can_pattern = r'([A-Z_]+)\.([A-Z_][A-Z0-9_]*[A-Z0-9])\.([A-Za-z_][A-Za-z0-9_]*)'
    
    signals = []
    matches = re.finditer(can_pattern, text_content)
    
    for match in matches:
        domain = match.group(1)
        signal_name = match.group(2)
        parameter = match.group(3)
        
        signal_info = {
            'full_name': match.group(0),
            'domain': domain,
            'signal_name': signal_name,
            'parameter': parameter,
            'hierarchy_level': determine_hierarchy_level(domain),
            'signal_type': determine_signal_type(signal_name),
            'values': determine_possible_values(text_content, match.group(0))
        }
        
        signals.append(signal_info)
    
    return signals

def determine_possible_values(text, signal_name):
    """
    Determine possible values for CAN signal
    """
    # Common automotive signal value patterns
    value_patterns = {
        'binary': r'(Active|Inactive|ON|OFF|0|1|Present|Absent)',
        'status': r'(Ready|Not Ready|OK|Error|Valid|Invalid)',
        'state': r'(Sleep|Suspend|Active|Standby|Shutdown)'
    }
    
    found_values = []
    for pattern_type, pattern in value_patterns.items():
        matches = re.findall(pattern, text)
        if matches:
            found_values.extend(matches)
    
    # Default to binary if no specific values found
    if not found_values:
        found_values = ['0', '1']
    
    return list(set(found_values))  # Remove duplicates
```

def validate_flow_logic(analysis):
    """
    Validate logical flow from start to end
    """
    checks = {
        'has_start_terminator': check_start_terminator(analysis),
        'has_end_terminator': check_end_terminator(analysis),
        'all_blocks_connected': check_block_connectivity(analysis),
        'decision_paths_valid': check_decision_paths(analysis),
        'no_orphaned_blocks': check_for_orphans(analysis),
        'cycles_properly_documented': check_cycle_documentation(analysis)
    }
    
    passed_checks = sum(checks.values())
    total_checks = len(checks)
    score = (passed_checks / total_checks) * 100
    
    return {
        'score': score,
        'checks': checks,
        'passed': score >= 95.0
    }

def validate_decision_points(analysis):
    """
    Validate decision point logic and completeness
    """
    if 'decision_points' not in analysis:
        return {'score': 100, 'checks': {}, 'passed': True}
    
    checks = {}
    for i, decision in enumerate(analysis['decision_points']):
        decision_checks = {
            f'decision_{i+1}_has_condition': bool(decision.get('condition_text')),
            f'decision_{i+1}_has_yes_path': bool(decision.get('paths', {}).get('yes_path')),
            f'decision_{i+1}_has_no_path': bool(decision.get('paths', {}).get('no_path')),
            f'decision_{i+1}_logic_documented': bool(decision.get('logical_operators')),
            f'decision_{i+1}_error_handling': bool(decision.get('error_handling'))
        }
        checks.update(decision_checks)
    
    passed_checks = sum(checks.values())
    total_checks = len(checks)
    score = (passed_checks / total_checks) * 100 if total_checks > 0 else 100
    
    return {
        'score': score,
        'checks': checks,
        'passed': score >= 90.0
    }
